======================
Preprocessor Condition
======================
#if exists ( HOME) && file_exists ("/full/path//to/file/.bin"	) || !a && b
#endif
----------------------
(source_file
 (preprocessor_if
  condition: (or
   left: (and
    left: (exists (environment_variable (identifier)))
    right: (file_exists filename: (filename))
   )
   right: (and
	left: (not (identifier))
    right: (identifier)
   )
  )
 )
 (preprocessor_endif)
)

)
